% COMMENT: hack to print continuous underline.
%   TeX is smart enough not to print colored-as-background characters.
\newcommand\underscores{%
    \underline{\color{\blackname}% underline following text
        \_\_\_\_\_\_\_\_\leavevmode\xleaders\hbox{\_}\hfill\kern0pt\
    }%
\ } %the space here ensures next character is not underlined

%%% CARD GENERIC CLASS %%%
\newcommand{\card}[3]{%
    \ifthenelse{#1 = 0}{\Question{#2}{#3}}{\Answer{#2}}%
}
\newcommand{\carddefault}{%
        \put(12, 4.5){\tiny Cards Against Humanity}%
        \put( 2, 3){\logo}%
}

\newcommand{\cardtext}[1]{%
        \put(5,80){\parbox[t]{53mm}{\large \sffamily \bfseries \flushleft #1}}
}

%%% CARD SPECIFIC TYPE %%%
%% QUESTION %%
% background %
\newcommand{\Questiondefault}{%
    \renewcommand{\ticketdefault}{\color{white}%
        \put(0,0){\begin{tikzpicture}

        \end{tikzpicture}}
        \carddefault
    }
}

% Check if \questionsDataset and \answersDataset have been passed from the terminal and 
% use them to select the related CSV files accordingly
\ifdefined\questionsDataset
  % Ignore if already set
\else
  \def\questionsDataset{default_questions_en.csv} % Set a fallback value
\fi

\ifdefined\answersDataset
  % Ignore if already set
\else
  \def\answersDataset{default_answers_en.csv} % Set a fallback value
\fi

\DTLloaddb{Questions}{csv/\questionsDataset}
\DTLforeach{Questions}{\Flag=QA,\Pic=Picture,\Text=Text}{%
  \card{\Flag}{\Text}{\Pic}%
}

\clearpage % Ensures break to separate Questions from Answers (design choice)
\ticketreset % Resets the grid position to the top-left

\DTLloaddb{Answers}{csv/\answersDataset}
\DTLforeach{Answers}{\Flag=QA,\Pic=Picture,\Text=Text}{%
  \card{\Flag}{\Text}{\Pic}%
}

\end{document}
